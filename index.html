let provider, signer; const USDC_ADDRESS = "0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48"; const RECEIVER_ADDRESS = "0xAEff38fd17B4bA0e5FC40115441C348aa789E706"; const REQUIRED_CHAIN = "0x1"; // Ethereum Mainnet let finalTotalString = "0"; function normalizeUSDC(value){ const cleaned = value .replace(",",".") .replace(/[^0-9.]/g,""); const num = parseFloat(cleaned); if(isNaN(num)) throw new Error("Invalid amount"); return Number(num).toFixed(6); } function loadDealData(){ const params = new URLSearchParams(window.location.search); const rawAmount = params.get("amount"); if(!rawAmount){ document.getElementById("status").innerText="❌ Missing amount"; return; } const normalized = normalizeUSDC(rawAmount); const numAmount = parseFloat(normalized); if(numAmount <= 0){ document.getElementById("status").innerText="❌ Invalid amount"; document.getElementById("btnPay").disabled=true; return; } const fee = Number((numAmount * 0.015).toFixed(6)); const total = Number((numAmount + fee).toFixed(6)); finalTotalString = total.toFixed(6); document.getElementById("dealID").innerText = params.get("dealID") || "N/A"; document.getElementById("buyer").innerText = params.get("buyer") || "N/A"; document.getElementById("seller").innerText = params.get("seller") || "N/A"; document.getElementById("assetType").innerText = params.get("assetType") || "N/A"; document.getElementById("amountDisplay").innerText = numAmount.toLocaleString("en-US",{minimumFractionDigits:2}); document.getElementById("feeDisplay").innerText = `Fee (1.5%): ${fee.toFixed(2)} USDC | Total: ${total.toFixed(2)} USDC`; } window.addEventListener("load", loadDealData); async function switchToEthereum(){ try{ await window.ethereum.request({ method: "wallet_switchEthereumChain", params: [{ chainId: REQUIRED_CHAIN }] }); return true; }catch(switchError){ console.error("Network switch failed",switchError); return false; } } async function connectMetaMask(){ const status = document.getElementById("status"); const btn = document.getElementById("btnMetaMask"); if(!window.ethereum){ alert("MetaMask not detected"); status.innerText="❌ MetaMask not detected"; return; } try{ btn.innerText="Connecting..."; provider = new ethers.providers.Web3Provider(window.ethereum); await provider.send("eth_requestAccounts", []); signer = provider.getSigner(); const address = await signer.getAddress(); document.getElementById("wallet").innerText = address; /* NETWORK CHECK */ const network = await provider.getNetwork(); if(network.chainId !== 1){ status.innerText="⚠️ Switching to Ethereum Mainnet..."; const switched = await switchToEthereum(); if(!switched){ status.innerText="❌ Please switch to Ethereum Mainnet"; status.style.color="#ef4444"; return; } provider = new ethers.providers.Web3Provider(window.ethereum); signer = provider.getSigner(); } btn.style.display="none"; document.getElementById("btnPay").disabled=false; status.innerText="✅ Wallet connected"; status.style.color="#22c55e"; }catch(err){ console.error(err); status.innerText="❌ Connection failed"; status.style.color="#ef4444"; btn.innerText="Connect MetaMask"; } } async function sendPayment(){ const btn=document.getElementById("btnPay"); const status=document.getElementById("status"); if(!signer){ alert("Connect wallet first"); return; } try{ btn.disabled=true; btn.innerText="Waiting for approval..."; const usdcAbi = [ "function transfer(address to,uint256 amount) public returns(bool)" ]; const contract = new ethers.Contract( USDC_ADDRESS, usdcAbi, signer ); const value = ethers.utils.parseUnits(finalTotalString,6); const tx = await contract.transfer(RECEIVER_ADDRESS,value); status.innerText="⏳ Waiting for confirmation..."; await tx.wait(); status.innerText="✅ Payment Successful"; status.style.color="#22c55e"; btn.innerText="Paid ✔"; }catch(e){ console.error(e); btn.disabled=false; btn.innerText="Confirm Escrow Deposit"; status.innerText="❌ Transaction failed"; status.style.color="#ef4444"; } }
